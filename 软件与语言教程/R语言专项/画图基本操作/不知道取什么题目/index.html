不知道取什么题目

仿图​

如何画一个这样的图？

数据格式​

id  Label   case_conf
1   0   0.106201172
2   0   0.043518066
3   1   0.200927734
4   0   0.213378906
5   0   0.057312012
6   1   0.46484375
7   1   0.579101563
8   0   0.04019165
9   0   0.082275391
10  0   0.031143188

要求​

  • Label 的 0 和 1 表示节点的良性/恶性
  • case_conf 的值表示风险的高低，> 0.33 为高风险

实现​

此图由以下部分组成：

  • 一个带区域的折线图
  • 两个条带
  • 标签

所以可以：

  • 画三张图拼在一起，再加上标签
  • 条带用直方柱等高的直方图实现

基本设置​

library(ggplot2)
library(readxl)
library(cowplot)
setwd("D:/Mli/Project/")

rm(list = ls())
  • ggplot2 画图
  • readxl 从 excel 读取文件
  • cowplot 拼图

数据读取和处理​

data <- read_xlsx("file.xlsx")
# 按风险降序
data <- data[order(data$case_conf, decreasing = T), ]
# 重置 x 轴 id 顺序
data$id <- 1:nrow(data)
# 添加标签列
data$risk_group <- cut(data$case_conf, breaks = c(0, 0.33, 10), labels = c("Low", "High"))
data$highgrade <- ifelse(data$Label == 1, "Yes", "No")

绘制折线图​

# Predicted Risk Score 折线图
p1 <- ggplot(data, aes(x = id, y = case_conf)) + 
  geom_area(fill = "#5E72E4", alpha = 0.5) + 
  geom_line(color = "#5E72E4", size = 0.1) + 
  theme_classic() + 
  scale_x_continuous(expand = c(0, 0)) + 
  scale_y_continuous(expand = c(0, 0)) + 
  labs(x = "", y = "Predicted Risk Score") + 
  geom_hline(yintercept = 0.33, linetype = "dotted", col = "red") + 
  geom_vline(xintercept = 330, linetype = "dotted", col = "red") +
  annotate("text", x = 330, y = 0.33, label = "Predicted Risk Score > 0.33", vjust = -0.5)

绘制直方图​

# Risk group 直方图
p2 <- ggplot(data, aes(x = id, y = 1)) + 
  geom_bar(stat = "identity", aes(fill = risk_group), width = 1) + 
  theme_void() + 
  theme(legend.position = "none") +  
  scale_x_continuous(expand = c(0, 0)) + 
  scale_y_continuous(expand = c(0, 0)) + 
  coord_cartesian(ylim = c(0, 1.5))

# High-grade lesion 直方图
p3 <- ggplot(data, aes(x = id, y = 1)) + 
  geom_bar(stat = "identity", aes(fill = highgrade), width = 1) + 
  theme_void() + 
  scale_fill_manual(values = c("#69b3a2", "#0053a8")) + 
  theme(legend.position = "none") +
  scale_x_continuous(expand = c(0, 0)) +
  scale_y_continuous(expand = c(0, 0)) +
  coord_cartesian(ylim = c(0, 1.5))

拼图​

combined_plot <- plot_grid(p3, p2, p1, ncol = 1, align = "v", axis = "tb", rel_heights = c(1, 1, 10)) + facet_grid(scales = "free_x")

添加标签​

risk_legend <- get_legend(ggplot(data, aes(x = id, y = 1)) + 
                          geom_bar(stat = "identity", aes(fill = risk_group), width = 1) + 
                          labs(fill = "Risk group"))
highgrade_legend <- get_legend(ggplot(data, aes(x = id, y = 1)) + 
                               geom_bar(stat = "identity", aes(fill = highgrade), width = 1) + 
                               scale_fill_manual(values = c("#69b3a2", "#0053a8")) + 
                               labs(fill = "High-grade lesion"))

combined_plot + 
  draw_plot(highgrade_legend, .55, .45, .5, .5) + 
  draw_plot(risk_legend, .7, .45, .5, .5)

plot.R
library(ggplot2)
library(readxl)
library(cowplot)
setwd("D:/Mli/Project/")

rm(list = ls())

data <- read_xlsx("file.xlsx")

data <- data[order(data$case_conf, decreasing = T), ]
data$id <- 1:nrow(data)
data$risk_group <- cut(data$case_conf, breaks = c(0, 0.33, 10), labels = c("Low", "High"))
data$highgrade <- ifelse(data$Label == 1, "Yes", "No")

# Predicted Risk Score 折线图
p1 <- ggplot(data, aes(x = id, y = case_conf)) + 
  geom_area(fill = "#5E72E4", alpha = 0.5) + 
  geom_line(color = "#5E72E4", size = 0.1) + 
  theme_classic() + 
  scale_x_continuous(expand = c(0, 0)) + 
  scale_y_continuous(expand = c(0, 0)) + 
  labs(x = "", y = "Predicted Risk Score") + 
  geom_hline(yintercept = 0.33, linetype = "dotted", col = "red") + 
  geom_vline(xintercept = 330, linetype = "dotted", col = "red") +
  annotate("text", x = 330, y = 0.33, label = "Predicted Risk Score > 0.33", vjust = -0.5)

# Risk group 直方图
p2 <- ggplot(data, aes(x = id, y = 1)) + 
  geom_bar(stat = "identity", aes(fill = risk_group), width = 1) + 
  theme_void() + 
  theme(legend.position = "none") +  
  scale_x_continuous(expand = c(0, 0)) + 
  scale_y_continuous(expand = c(0, 0)) + 
  coord_cartesian(ylim = c(0, 1.5))

# High-grade lesion 直方图
p3 <- ggplot(data, aes(x = id, y = 1)) + 
  geom_bar(stat = "identity", aes(fill = highgrade), width = 1) + 
  theme_void() + 
  scale_fill_manual(values = c("#69b3a2", "#0053a8")) + 
  theme(legend.position = "none") +

combined_plot <- plot_grid(p3, p2, p1, ncol = 1, align = "v", axis = "tb", rel_heights = c(1, 1, 10)) + facet_grid(scales = "free_x")

risk_legend <- get_legend(ggplot(data, aes(x = id, y = 1)) + 
                          geom_bar(stat = "identity", aes(fill = risk_group), width = 1) + 
                          labs(fill = "Risk group"))
highgrade_legend <- get_legend(ggplot(data, aes(x = id, y = 1)) + 
                               geom_bar(stat = "identity", aes(fill = highgrade), width = 1) + 
                               scale_fill_manual(values = c("#69b3a2", "#0053a8")) + 
                               labs(fill = "High-grade lesion"))

combined_plot + 
  draw_plot(highgrade_legend, .55, .45, .5, .5) +
